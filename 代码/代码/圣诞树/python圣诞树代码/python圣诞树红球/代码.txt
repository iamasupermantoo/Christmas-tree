def koc(x, y, size):
  pensize(3)
  pencolor("orange")
  penup()
  goto(x, y)
  pendown()
  begin_fill()
  fillcolor('yellow')
  for i in range(5):
      left(72)
      fd(size)
      right(144)
      fd(size)
  end_fill()
